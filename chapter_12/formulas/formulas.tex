\begin{tikzpicture}

        \begin{enumerate}
            \setcounter{enumi}{4}
            \item $\abs{\fvec{v}} = \sqrt{v_1^2+v_2^2+v_3^2}$\\
                $\abs{\fvec{v}} = \sqrt{(x_1-x_2)^2+(y_1-y_2)^2+(z_1-z_2)^2}$
            \item $\fvec{u} + \fvec{v} = \langle u_1 + v_1, u_2 + v_2, u_3 + v_3 \rangle$
            \item $k\fvec{u} = \langle ku_1, ku_2, ku_3 \rangle$
            \item $\dotp{u}{v} = u_1v_1 + u_2v_2 + u_3v_3$ \\
                $\dotp{u}{v} = \abs{\fvec{u}} \abs{\fvec{v}} cos \theta$
            \item $proj_v\fvec{u} = (\frac{\dotp{u}{v}}{\abs{\fvec{v}}^2})\fvec{v} = (\frac{\dotp{u}{v}}{\abs{\fvec{v}}})\frac{\fvec{v}}{\abs{\fvec{v}}}$
            \item $\abs{\fvec{u}}cos \theta = \frac{\dotp{u}{v}}{\abs{\fvec{v}}} = \fvec{u} \bdot \frac{\fvec{v}}{\abs{\fvec{v}}}$
            \item $\xp{u}{v} = (\abs{\fvec{u}}\abs{\fvec{v}} sin \theta) \uv{n}$ \\
                $\xp{u}{v} = (u_2v_3-u_3v_2)\iv - (u_1v_3-u_3v_1)\jv + (u_1v_2-u_2v_1)\kv$ \\
                $\xp{u}{v} = \threemat{\iv}{\jv}{\kv}{u_1}{u_2}{u_3}{v_1}{v_2}{v_3}$
            \item $\abs{\xp{u}{v}} = \abs{\fvec{u}}\abs{\fvec{v}} sin \theta = A$ \\
                $A$ is area of the parallelogram defined by $\fvec{u}$ and $\fvec{v}$\\
                $\frac{A}{2}$ is area of triangle defined by $\fvec{u}$ and $\fvec{v}$
        \end{enumerate}
    \end{minipage}

%------------ Properties ---------------
\begin{tikzpicture}
\node [formulabox] (box){%
    \begin{minipage}{0.7\textwidth}
        \textbf{Vector Algebra}
        \begin{enumerate}
            \item $\fvec{u} + \fvec{v} = \fvec{v} + \fvec{u}$
            \item ($\fvec{u} + \fvec{v}) + \fvec{w} = \fvec{u} + (\fvec{v} + \fvec{w})$
            \item $\fvec{u} + 0 = \fvec{u}$
            \item $\fvec{u} + (- \fvec{u}) = 0$
            \item $0\fvec{u} = 0$
            \item $1\fvec{u} = \fvec{u}$
            \item $a(b\fvec{u}) = (ab)\fvec{u}$
            \item $a(\fvec{u} + \fvec{v}) = a\fvec{u} + a \fvec{v}$
            \item $(a+b)\fvec{u} = a\fvec{u} + b\fvec{u}$
        \end{enumerate}
        \textbf{Unit Vectors}
        \begin{enumerate}
            \item $\abs{\uv{v}} = 1$
            \item $\iv = \langle 1,0,0 \rangle, \jv = \langle 0,1,0 \rangle, \kv = \langle 0,0,1 \rangle$
            \item $\fvec{u} = u_1\iv + u_2\jv +u_3\kv$
            \item $\frac{\fvec{v}}{\abs{\fvec{v}}}, \fvec{v} \neq 0$ is \textit{direction} of $\fvec{v}$
            \item $\abs{\frac{\fvec{v}}{\abs{\fvec{v}}}} = 1, \fvec{v} \neq 0$
            \item $\fvec{v} = \abs{\fvec{v}}\frac{\fvec{v}}{\abs{\fvec{v}}}, \fvec{v} \neq 0$
            \item $\dotp{i}{j} = \dotp{j}{k} = \dotp{i}{k} = 0$
            \item $\iv \times \iv = \jv \times \jv = \kv \times \kv = 0$
            \item $\iv \times \jv = -(\jv \times \iv) = \kv$
            \item $\jv \times \kv = -(\kv \times \jv) = \iv$
            \item $\kv \times \iv = -(\iv \times \kv) = \jv$
        \end{enumerate}
        \textbf{Dot Product}
        \begin{enumerate}
            \setcounter{enumi}{4}
            \item $\fvec{u} \perp \fvec{v} \iff \dotp{u}{v} = 0$
            \item $\dotp{u}{v} = \dotp{v}{u}$
            \item $(c\fvec{u}) \bdot \fvec{v} = \fvec{u} \bdot (c\fvec{v}) = c(\dotp{u}{v})$
            \item $\fvec{u} \bdot (\fvec{v} + \fvec{w}) = \dotp{u}{v} + \dotp{u}{w}$
            \item $\dotp{u}{u} = \abs{\fvec{u}}^2$
            \item $0 \bdot \fvec{u} = 0$
        \end{enumerate}
        \textbf{Cross Product}
        \begin{enumerate}
            \setcounter{enumi}{4}
            \item $\fvec{u} \parallel \fvec{v} \iff \xp{u}{v} = 0$
            \item $\overrightarrow{PQ} \times \overrightarrow{PR} \perp \mathcal{B}_{PQR}$ 
            \item $(r\fvec{u}) \times (s\fvec{v}) = (rs)(\xp{u}{v})$
            \item $\fvec{u} \times (\fvec{v} + \fvec{w}) = \xp{u}{v} + \xp{u}{w}$
            \item $\xp{v}{u} = -(\xp{u}{v})$
            \item $(\fvec{v} + \fvec{w}) \times \fvec{u} = \xp{v}{u} + \xp{w}{u}$
            \item $0 \times \fvec{u} = 0$
            \item $\fvec{u} \times (\xp{v}{w}) = (\dotp{u}{w})\fvec{v} - (\dotp{u}{v})\fvec{w}$
        \end{enumerate}
    \end{minipage}
};

\end{tikzpicture}

        \begin{itemize}
            \item Right hand rule: with thumb along positive $z$ axis, fingers curl from $x$ axis to $y$ axis
            \item Any vector of length 1 is a unit vector, but $\iv,\jv,\kv$ are standard
            \item $proj_v\fvec{u}$ is the projection of $\fvec{u}$ onto $\fvec{v}$
            \item $\fvec{v}$: vector
            \item $\overrightarrow{PQ}$: line from $P$ to $Q$
            \item $\uv{n}$: unit vector with direction $n$
            \item $\mathcal{B}, \mathcal{B}_{PQR}$: Arbitrary plane, plane containing points $P$, $Q$, and $R$
            \item $\mathbb{A}$: matrix
