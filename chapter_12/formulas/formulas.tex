%------------ Properties ---------------
\begin{tikzpicture}
\node [formulabox] (box){%
    \begin{minipage}{0.7\textwidth}
        \textbf{Vector Algebra}
        \begin{enumerate}
            \item $\fvec{u} + \fvec{v} = \fvec{v} + \fvec{u}$
            \item ($\fvec{u} + \fvec{v}) + \fvec{w} = \fvec{u} + (\fvec{v} + \fvec{w})$
            \item $\fvec{u} + 0 = \fvec{u}$
            \item $\fvec{u} + (- \fvec{u}) = 0$
            \item $0\fvec{u} = 0$
            \item $1\fvec{u} = \fvec{u}$
            \item $a(b\fvec{u}) = (ab)\fvec{u}$
            \item $a(\fvec{u} + \fvec{v}) = a\fvec{u} + a \fvec{v}$
            \item $(a+b)\fvec{u} = a\fvec{u} + b\fvec{u}$
        \end{enumerate}
        \textbf{Dot Product}
        \begin{enumerate}
            \setcounter{enumi}{4}
            \item $\fvec{u} \perp \fvec{v} \iff \dotp{u}{v} = 0$
            \item $\dotp{u}{v} = \dotp{v}{u}$
            \item $(c\fvec{u}) \bdot \fvec{v} = \fvec{u} \bdot (c\fvec{v}) = c(\dotp{u}{v})$
            \item $\fvec{u} \bdot (\fvec{v} + \fvec{w}) = \dotp{u}{v} + \dotp{u}{w}$
            \item $\dotp{u}{u} = \abs{\fvec{u}}^2$
            \item $0 \bdot \fvec{u} = 0$
        \end{enumerate}
        \textbf{Cross Product}
        \begin{enumerate}
            \setcounter{enumi}{4}
            \item $\fvec{u} \parallel \fvec{v} \iff \xp{u}{v} = 0$
            \item $(r\fvec{u}) \times (s\fvec{v}) = (rs)(\xp{u}{v})$
            \item $\fvec{u} \times (\fvec{v} + \fvec{w}) = \xp{u}{v} + \xp{u}{w}$
            \item $\xp{v}{u} = -(\xp{u}{v})$
            \item $(\fvec{v} + \fvec{w}) \times \fvec{u} = \xp{v}{u} + \xp{w}{u}$
            \item $0 \times \fvec{u} = 0$
            \item $\fvec{u} \times (\xp{v}{w}) = (\dotp{u}{w})\fvec{v} - (\dotp{u}{v})\fvec{w}$
        \end{enumerate}
    \end{minipage}
};

\end{tikzpicture}
